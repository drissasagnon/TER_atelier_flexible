cmake_minimum_required(VERSION 2.8.3)
project(commande_locale)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED COMPONENTS
	roscpp
	std_msgs
	sensor_msgs
	image_transport
	cv_bridge
	genmsg
	message_generation
	shuttles
	aiguillages)

#set(EXECUTABLE_OUTPUT_PATH ../../../)

	FILES

	Msg_AddProduct.msg

	Actionneurs.msg

)
